 19: /**
 20:  * Tag object
 21:  *
 22:  * This class extends {@link OntologyObject} to implement a concrete tag object class.
 23:  *
 24:  * A tag object is used to <i>identify</i>, <i>document</i> and <i>share</i> a data
 25:  * property, it represents the <i>metadata</i> of a data property, or its <i>definition</i>
 26:  * in an ontology.
 27:  *
 28:  * This class features a {@link kTAG_TERMS} offset which is an array of {@link TermObject}
 29:  * references representing a path of vertex and predicate terms of the ontology graph, this
 30:  * chain of elements represents a lexical construct, like a phrase, that provides the
 31:  * explanation or definition of the current object.
 32:  *
 33:  * Using a chain of terms, rather than a single term, to provide metadata for a data element
 34:  * gives much more flexibility and reusability when defining a data dictionary. This
 35:  * sequence is divided in three main sections:
 36:  *
 37:  * <ul>
 38:  *  <li><i>Feature</i>: The first vertex represents the data element feature or trait, this
 39:  *      term defines <em>what the data element is</em>.
 40:  *  <li><i>Method</i>: The vertices between the first and the last ones represent the
 41:  *      <em>methodology</em> or <em>method</em> by which the described data was obtained.
 42:  *  <li><i>Scale</i>: The last vertex of the path represents the <em>unit</em> or
 43:  *      <em>scale</em> in which the described value is expressed in.
 44:  * </ul>
 45:  *
 46:  * The concatenation of these term references, separated by the
 47:  * {@link kTOKEN_INDEX_SEPARATOR} becomes the object global identifier which is stored in
 48:  * the object's native identifier offset.
 49:  *
 50:  * Tags have another offset, {@link kTAG_SEQ}, which is an integer sequence number: this
 51:  * value must be unique within the tags domain of the current ontology. Unlike global
 52:  * identifiers, this value may change across implementations, but this is the value used to
 53:  * uniquely identify tags among the other elements of the ontology and database.
 54:  *
 55:  * <em>All offsets in all classes, including this one, are tag sequence numbers, which makes
 56:  * the Tag class key in the structure and behaviour of all the elements implemented in this
 57:  * library</em>.
 58:  *
 59:  * The class features the following default offsets:
 60:  *
 61:  * <ul>
 62:  *  <li><tt>{@link kTAG_NID}</tt>: <em>Native identifier</em>. This required attribute holds
 63:  *      a string value which represents the global identifier of the current object. This
 64:  *      identifier is immutable and represents the unique key. This string is constituted by
 65:  *      the concatenation of all term references stored in the current object's branch,
 66:  *      {@link kTAG_TERMS}. This attribute must be managed with its offset; in derived
 67:  *      classes it will be automatically assigned end only available as read-only.
 68:  *  <li><tt>{@link kTAG_SEQ}</tt>: <em>Sequence</em>. This required attribute holds
 69:  *      an integer value which represents the current object's sequence number, as with the
 70:  *      global identifier, this value must be unique, except that it may change across
 71:  *      implementations. All offset keys in all objects derived from this class ancestor are
 72:  *      references to this sequence number. This attribute must be managed with its offset;
 73:  *      in derived classes it will be automatically assigned end only available as
 74:  *      read-only.
 75:  *  <li><tt>{@link kTAG_TERMS}</tt>: <em>Branch</em>. This required attribute holds
 76:  *      the list of terms comprising the current tag: this is an array of term references
 77:  *      provided as an odd sequence of vertices and predicates forming a path of the
 78:  *      ontology graph in which the first vertex defines the feature, the middle ones
 79:  *      define the methodology and the last element indicates the scale or unit. To populate
 80:  *      the elements of this path, use the {@link TermPush()} amd {@link TermPop()}
 81:  *      offset accessor methods which respectively add and remove elements of the branch as
 82:  *      if it was a stack.
 83:  *  <li><tt>{@link kTAG_DATA_TYPE}</tt>: <em>Data type</em>. This attribute is an enumerated
 84:  *      set of values listing all the <em>data types</em> that the value of the property
 85:  *      defined by the current tag may take. To populate and handle individual data types
 86:  *      use the {@link DataType()} offset accessor method. This property is
 87:  *      <em>required</em> by all tag objects.
 88:  *  <li><tt>{@link kTAG_DATA_KIND}</tt>: <em>Data kind</em>. This attribute is an enumerated
 89:  *      set of values providing the <em>data attributes</em> of the property defined by the
 90:  *      current tag. This may be whether the property is a list of values, or if the
 91:  *      property is required or not. To populate and handle individual data kinds use the
 92:  *      {@link DataKind()} offset accessor method.
 93:  *  <li><tt>{@link kTAG_LABEL}</tt>: <em>Label</em>. The label represents the <i>name or
 94:  *      short description</i> of the data property that the current object defines. All tags
 95:  *      <em>should</em> have a label, since this is how human users will be able to identify
 96:  *      and select them. This attribute has the {@link kTYPE_KIND_VALUE} data type, which
 97:  *      is constituted by a list of elements in which the {@link kTAG_SUB_LANGUAGE} item holds
 98:  *      the label language code and the {@link kTAG_SUB_TEXT} holds the label text. To
 99:  *      populate and handle labels by language, use the {@link Label()} offset accessor
100:  *      method.
101:  *  <li><tt>{@link kTAG_DESCRIPTION}</tt>: <em>Description</em>. The description tag
102:  *      represents the <i>description or extended definition</i> of the property that the
103:  *      current tag defines. The description is similar to the <em>definition</em>, except
104:  *      that while the definition provides a description of the object it defines unrelated
105:  *      to context, the description adds to the definition the elements added by the current
106:  *      context. All tags <em>should</em> have a description, if the tag label is not enough
107:  *      to provide a sufficient description or definition. Descriptions have the
108:  *      {@link kTYPE_KIND_VALUE} data type, which is constituted by a list of elements in
109:  *      which the {@link kTAG_SUB_LANGUAGE} item holds the description language code and the
110:  *      {@link kTAG_SUB_TEXT} holds the description text. To populate and handle
111:  *      descriptions by language, use the {@link Description()} offset accessor method.
112:  * </ul>
113:  *
114:  * The {@link __toString()} method will return the value stored in the global identifier,
115:  * if set, or the computed global identifier which is the concatenation of all term
116:  * references stored in the {@link kTAG_TERMS} offset, separated by the
117:  * {@link kTOKEN_INDEX_SEPARATOR}.
118:  *
119:  * The class also features a pair of methods, {@link Feature()} and {@link Scale()}, which
120:  * respectively return the first and last vertex of the ontology branch this object
121:  * represents: the feature is a term that represents the main feature or trait that the
122:  * current tag defines, the scale is a term that represents the scale or unit in which
123:  * values defined by the current tag are expressed in.
124:  *
125:  * Objects of this class can hold any additional attribute that is considered necessary or
126:  * useful to define and share the current tag. In this class we define only those
127:  * attributes that constitute the core functionality of the object, derived classes will add
128:  * attributes specific to the domain in which the object will operate and eventually
129:  * restrict the functionality of other offsets to provide referential integrity.
130:  *
131:  *  @author     Milko A. Škofič <m.skofic@cgiar.org>
132:  *  @version    1.00 05/02/2014
133:  */

151:      * <h4>Return global identifier</h4>
152:      *
153:      * The global identifier of tags is stored in its {@link kTAG_NID} offset: if set, this
154:      * method will return that value. If that offset is not set, the method will concatenate
155:      * the value of all the elements of the {@link kTAG_TERMS} separated by the
156:      * {@link kTOKEN_INDEX_SEPARATOR} token.
157:      *
158:      * If the {@link kTAG_TERMS} offset is not set, the method will return an empty string.
159:      *
160:      * @access public
161:      * @return string               The global identifier.
162:      */

164:     {
165:         //
166:         // Get native identifier.
167:         //
168:         if( \ArrayObject::offsetExists( kTAG_NID ) )
169:             return \ArrayObject::offsetGet( kTAG_NID );                             // ==>
170:         
171:         //
172:         // Compute global identifier.
173:         //

175:             return implode( kTOKEN_INDEX_SEPARATOR,
176:                             \ArrayObject::offsetGet( kTAG_TERMS ) );                // ==>
177:

421:     /**
422:      * Add to terms path
423:      *
424:      * This method can be used to append elements to the object's terms path, it will add
425:      * the provided element to the end of the path.
426:      *
427:      * If you provide a {@link TermObject} as the parameter, the method will
428:      * {@link Reference()} it.
429:      *
430:      * The method will return the number of elements in the path.
431:      *
432:      * @param mixed                 $theTerm            Term reference or object.
433:      *
434:      * @access public
435:      * @return integer              Number of elements in path.
436:      *
437:      * @see kTAG_TERMS
438:      *
439:      * @uses manageElementMatchOffset()
440:      */
441:     public function TermPush( $theTerm )
442:     {
443:         //
444:         // Handle objects.
445:         //
446:         if( is_object( $theTerm ) )
447:         {
448:             //
449:             // If term, get its reference.
450:             //
451:             if( $theTerm instanceof TermObject )
452:                 $theTerm = $theTerm->Reference();
453:         
454:             //
455:             // If not a term, complain.
456:             //
457:             else
458:                 throw new \Exception(
459:                     "Unable to add element to terms path: "
460:                    ."provided an object other than term." );                    // !@! ==>
461:     
462:         } // Object.
463:     
464:         //
465:         // Cast to setring.
466:         //
467:         else
468:             $theTerm = (string) $theTerm;
469:         
470:         //
471:         // Get current path.
472:         //
473:         $path = ( \ArrayObject::offsetExists( kTAG_TERMS ) )
474:               ? \ArrayObject::offsetGet( kTAG_TERMS )
475:               : Array();
476:         
477:         //
478:         // Add element.
479:         //
480:         $path[] = $theTerm;
481:         
482:         //
483:         // Get count.
484:         //
485:         $count = count( $path );
486:         
487:         //
488:         // Set offset.
489:         //
490:         $this->offsetSet( kTAG_TERMS, $path );
491:         
492:         return $count;                                                              // ==>
493:     
494:     } // TermPush.

501:     /**
502:      * Pop terms from path
503:      *
504:      * This method can be used to pop elements off the end of the object's terms path, it
505:      * will remove the last element in the sequence.
506:      *
507:      * When you remove the last element of the path, the method will also remove the offset.
508:      *
509:      * The method will return the removed element; if the path is empty, the method will
510:      * return <tt>NULL</tt>.
511:      *
512:      * @access public
513:      * @return string               Removed element or <tt>NULL</tt>.
514:      *
515:      * @see kTAG_TERMS
516:      *
517:      * @uses manageElementMatchOffset()
518:      */
519:     public function TermPop()
520:     {
521:         //
522:         // Get current path.
523:         //
524:         if( \ArrayObject::offsetExists( kTAG_TERMS ) )
525:         {
526:             //
527:             // Get current path.
528:             //
529:             $path = \ArrayObject::offsetGet( kTAG_TERMS );
530:             
531:             //
532:             // Pop element.
533:             //
534:             $element = array_pop( $path );
535:             
536:             //
537:             // Update parh.
538:             //
539:             if( count( $path ) )
540:                 $this->offsetSet( kTAG_TERMS, $path );
541:             
542:             //
543:             // Delete offset.
544:             //
545:             else
546:                 $this->offsetUnset( kTAG_TERMS );
547:             
548:             return $element;                                                        // ==>
549:         
550:         } // Has branch.
551:         
552:         return NULL;                                                                // ==>
553:     
554:     } // TermPop.
