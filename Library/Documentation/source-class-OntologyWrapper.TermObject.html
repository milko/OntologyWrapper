 19: /**
 20:  * Term object
 21:  *
 22:  * This class extends {@link OntologyObject} to implement a concrete term object class.
 23:  *
 24:  * A term object holds the necessary information to <i>uniquely identify</i>,
 25:  * <i>document</i> and <i>share</i> a <i>generic term or concept</i> which is <i>not related
 26:  * to a specific context</i>.
 27:  *
 28:  * For instance, a <tt>name</tt> is defined as a string or text that identifies something,
 29:  * this is true for both a person name or an object name, however, the term <tt>name</tt>
 30:  * will bare a different meaning depending on what context it is used in: the term object
 31:  * holds the definition of that will not change with its context.
 32:  *
 33:  * The class features the following default offsets:
 34:  *
 35:  * <ul>
 36:  *  <li><tt>{@link kTAG_NID}</tt>: <em>Native identifier</em>. This required attribute holds
 37:  *      the term global identifier. By convention this value is the combination of the
 38:  *      namespace, {@link kTAG_NS}, and the local identifier, {@link kTAG_LID}, separated by
 39:  *      the {@link kTOKEN_NAMESPACE_SEPARATOR} token. In practice, the global identifier may
 40:  *      be manually set. This attribute must be managed with its offset.
 41:  *  <li><tt>{@link kTAG_NS}</tt>: <em>Namespace</em>. This optional attribute is a reference
 42:  *      to another term object that represents the namespace of the current term. It is by
 43:  *      definition the global identifier of the namespace term. This attribute must be
 44:  *      managed with its offset.
 45:  *  <li><tt>{@link kTAG_LID}</tt>: <em>Local identifier</em>. This required attribute is a
 46:  *      string that represents the current term unique identifier within its namespace. The
 47:  *      combination of the current term's namespace and this attribute form the term's
 48:  *      global identifier. This attribute must be managed with its offset.
 49:  *  <li><tt>{@link kTAG_LABEL}</tt>: <em>Label</em>. The label represents the <i>name or
 50:  *      short description</i> of the term that the current object defines. All terms
 51:  *      <em>should</em> have a label, since this is how human users will be able to identify
 52:  *      and select them. This attribute has the {@link kTYPE_KIND_VALUE} data type, which
 53:  *      is constituted by a list of elements in which the {@link kTAG_SUB_LANGUAGE} item holds
 54:  *      the label language code and the {@link kTAG_SUB_TEXT} holds the label text. To
 55:  *      populate and handle labels by language, use the {@link Label()} offset accessor
 56:  *      method. Some terms may not have a language element, for instance the number
 57:  *      <tt>2</tt> may not need to be expressed in other ways.
 58:  *  <li><tt>{@link kTAG_DEFINITION}</tt>: <em>Definition</em>. The definition represents the
 59:  *      <i>description or extended definition</i> of the term that the current object object
 60:  *      defines. The definition is similar to the <em>description</em>, except that while
 61:  *      the description provides context specific information, the definition should not.
 62:  *      All terms <em>should</em> have a definition, if the object label is not enough to
 63:  *      provide a sufficient definition. Definitions have the {@link kTYPE_KIND_VALUE} data
 64:  *      type in which the {@link kTAG_SUB_LANGUAGE} element holds the definition language code
 65:  *      and the {@link kTAG_SUB_TEXT} holds the definition text. To populate and handle
 66:  *      definitions by language, use the {@link Definition()} offset accessor method.
 67:  * </ul>
 68:  *
 69:  * The {@link __toString()} method will return the value stored in the native identifier,
 70:  * if set, or the computed global identifier if at least the local identifier is set; if the
 71:  * latter is not set, the method will fail.
 72:  *
 73:  * Objects of this class can hold any additional attribute that is considered necessary or
 74:  * useful to define and share the current term. In this class we define only those
 75:  * attributes that constitute the core functionality of the object, derived classes will add
 76:  * attributes specific to the domain in which the object will operate.
 77:  *
 78:  *  @author     Milko A. Škofič <m.skofic@cgiar.org>
 79:  *  @version    1.00 07/02/2014
 80:  */

 97:     /**
 98:      * <h4>Return global identifier</h4>
 99:      *
100:      * If the native identifier, {@link kTAG_NID}, is set, this method will return its
101:      * value. If that offset is not yet set, the method will compute the global identifier
102:      * by concatenating the object's namespace, {@link kTAG_NS}, with the object's local
103:      * identifier, {@link kTAG_LID}, separated by the {@link kTOKEN_NAMESPACE_SEPARATOR}
104:      * token. This will only occur if the object has the local identifier, if that is not
105:      * the case, the method will return an empty string to prevent the method from causing
106:      * an error.
107:      *
108:      * @access public
109:      * @return string               The global identifier.
110:      */
111:     public function __toString()
112:     {
113:         //
114:         // Get native identifier.
115:         //
116:         if( \ArrayObject::offsetExists( kTAG_NID ) )
117:             return \ArrayObject::offsetGet( kTAG_NID );                             // ==>
118:         
119:         //
120:         // Compute global identifier.
121:         //
122:         if( \ArrayObject::offsetExists( kTAG_LID ) )
123:             return ( \ArrayObject::offsetExists( kTAG_NS ) )
124:                  ? (\ArrayObject::offsetGet( kTAG_NS )
125:                    .kTOKEN_NAMESPACE_SEPARATOR
126:                    .\ArrayObject::offsetGet( kTAG_LID ))                            // ==>
127:                  : \ArrayObject::offsetGet( kTAG_LID );                             // ==>
128:         
129:         return '';                                                                  // ==>
130:     
131:     } // __toString.

148:      * Manage tag label
149:      *
150:      * This method can be used to add, retrieve and delete individual labels by language,
151:      * while managing all the labels as a whole can be done using the {@link kTAG_LABEL}
152:      * offset.
153:      *
154:      * The method expects the following parameters:
155:      *
156:      * <ul>
157:      *  <li><b>$theLanguage</b>: This parameter holds the language code of the text we
158:      *      want to manage. If <tt>NULL</tt>, it means that there is a text without a
159:      *      language; this can occur if the label is a name unrelated to any language.
160:      *  <li><b>$theValue</b>: This parameter identifies the label text or operation:
161:      *   <ul>
162:      *      <li><tt>NULL</tt>: Return the label of the provided language.
163:      *      <li><tt>FALSE</tt>: Delete the label of the provided language.
164:      *      <li><i>other</i>: Any other value is cast to string and interpreted as the text
165:      *          of the label in the provided language which will be inserted, or that will
166:      *          replace an existing entry.
167:      *   </ul>
168:      *  <li><tt>$getOld</tt>: Determines what the method will return:
169:      *   <ul>
170:      *      <li><tt>TRUE</tt>: Return the value <i>before</i> it was eventually modified.
171:      *      <li><tt>FALSE</tt>: Return the value <i>after</i> it was eventually modified.
172:      *   </ul>
173:      * </ul>
174:      *
175:      * The method will return either the label text, or <tt>NULL</tt> if the language was
176:      * not matched.
177:      *
178:      * @param string                $theLanguage        Label language.
179:      * @param mixed                 $theValue           Label text or operation.
180:      * @param boolean               $getOld             TRUE get old value.
181:      *
182:      * @access public
183:      * @return mixed                Old or new value.
184:      *
185:      * @see kTAG_LABEL kTAG_SUB_LANGUAGE kTAG_SUB_TEXT
186:      *
187:      * @uses manageElementMatchOffset()
188:      */
189:     public function Label( $theLanguage, $theValue = NULL, $getOld = FALSE )
190:     {
191:         return $this->manageElementMatchOffset(
192:                 kTAG_LABEL, kTAG_SUB_LANGUAGE, kTAG_SUB_TEXT,
193:                 $theLanguage, $theValue, $getOld );                                 // ==>
194:     
195:     } // Label.

208:      *
209:      * The method expects the following parameters:
210:      *
211:      * <ul>
212:      *  <li><b>$theLanguage</b>: This parameter holds the language code of the text we
213:      *      want to manage. The <tt>NULL</tt> value should only be used if the label is not
214:      *      related to a specific language.
215:      *  <li><b>$theValue</b>: This parameter identifies the definition text or operation:
216:      *   <ul>
217:      *      <li><tt>NULL</tt>: Return the definition of the provided language.
218:      *      <li><tt>FALSE</tt>: Delete the definition of the provided language.
219:      *      <li><i>other</i>: Any other value is cast to string and interpreted as the text
220:      *          of the definition in the provided language which will be inserted, or that
221:      *          will replace an existing entry.
222:      *   </ul>
223:      *  <li><tt>$getOld</tt>: Determines what the method will return:

226:      *      <li><tt>FALSE</tt>: Return the value <i>after</i> it was eventually modified.
227:      *   </ul>
228:      * </ul>
229:      *
230:      * The method will return either the definition text, or <tt>NULL</tt> if the language
231:      * was not matched.
232:      *
233:      * @param string                $theLanguage        Definition language.
234:      * @param mixed                 $theValue           Definition text or operation.
235:      * @param boolean               $getOld             TRUE get old value.
236:      *
237:      * @access public

239:      *
240:      * @see kTAG_DEFINITION kTAG_SUB_LANGUAGE kTAG_SUB_TEXT

242:      * @uses manageElementMatchOffset()

290:     /**
291:      * Handle offset and value before setting it
292:      *
293:      * In this class we cast the value of the namespace into a term reference, ensuring
294:      * that if an object is provided this is a term.
295:      *
296:      * @param reference             $theOffset          Offset reference.
297:      * @param reference             $theValue           Offset value reference.
298:      *
299:      * @access protected
300:      * @return mixed                <tt>NULL</tt> set offset value, other, return.
301:      *
302:      * @throws Exception
303:      *
304:      * @see kTAG_NS
305:      */
306:     protected function preOffsetSet( &$theOffset, &$theValue )
307:     {
308:         //
309:         // Call parent method to resolve offset.
310:         //
311:         $ok = parent::preOffsetSet( $theOffset, $theValue );
312:         if( $ok === NULL )
313:         {
314:             //
315:             // Intercept namespace.
316:             //
317:             if( $theOffset == kTAG_NS )
318:             {
319:                 //
320:                 // Handle objects.
321:                 //
322:                 if( is_object( $theValue ) )
323:                 {
324:                     //
325:                     // If term, get its reference.
326:                     //
327:                     if( $theValue instanceof self )
328:                         $theValue = $theValue->Reference();
329:                 
330:                     //
331:                     // If not a term, complain.
332:                     //
333:                     else
334:                         throw new \Exception(
335:                             "Unable to set namespace: "
336:                            ."provided an object other than term." );            // !@! ==>
337:             
338:                 } // Object.
339:             
340:                 //
341:                 // Cast to setring.
342:                 //
343:                 else
344:                     $theValue = (string) $theValue;
345:             
346:             } // Setting namespace.
347:             
348:         } // Passed preflight.
349:         
350:         return $ok;                                                                 // ==>
351:     
352:     } // preOffsetSet.
