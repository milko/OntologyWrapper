111:     /**
112:      * Resolve an identifier
113:      *
114:      * We first check if the current collection is connected, if that is not the case, we
115:      * raise an exception.
116:      *
117:      * @param mixed                 $theIdentifier      Object identifier.
118:      * @param mixed                 $theOffset          Offset.
119:      * @param boolean               $asObject           Return object if <tt>TRUE</tt>.
120:      *
121:      * @access public
122:      * @return mixed                Found object, array, or <tt>NULL</tt>.
123:      *
124:      * @throws Exception
125:      */
126:     public function resolve( $theIdentifier, $theOffset = kTAG_NID, $asObject = TRUE )
127:     {
128:         //
129:         // Check if connected.
130:         //
131:         if( $this->isConnected() )
132:         {
133:             //
134:             // Resolve offset.
135:             //
136:             $theOffset = (string) OntologyObject::ResolveOffset( $theOffset, TRUE );
137:             
138:             //
139:             // Match object.
140:             //
141:             $object = $this->mConnection->findOne( array( $theOffset => $theIdentifier ) );
142:             if( $object !== NULL )
143:             {
144:                 //
145:                 // Return array.
146:                 //
147:                 if( ! $asObject )
148:                     return $object;                                                 // ==>
149:                 
150:                 //
151:                 // Check class.
152:                 //
153:                 if( array_key_exists( kTAG_CLASS, $object ) )
154:                 {
155:                     //
156:                     // Save class.
157:                     //
158:                     $class = $object[ kTAG_CLASS ];
159:                     
160:                     return new $class( $this, $object );                            // ==>
161:                 
162:                 } // Has class.
163:             
164:                 throw new \Exception(
165:                     "Unable to resolve object: "
166:                    ."missing object class." );                                  // !@! ==>
167:             
168:             } // Found.
169:             
170:             return NULL;                                                            // ==>
171:         
172:         } // Connected.
173:             
174:         throw new \Exception(
175:             "Unable to resolve object: "
176:            ."connection is not open." );                                        // !@! ==>
177:     
178:     } // resolve.
