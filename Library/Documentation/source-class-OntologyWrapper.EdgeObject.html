 20: /**
 21:  * Edge object
 22:  *
 23:  * This class extends {@link OntologyObject} to implement a concrete edge object class.
 24:  *
 25:  * This class implements a <em>directed graph</em> by <em>relating a subject vertex</em>
 26:  * with an <em>object vertex</em> through a <em>predicate</em>, the direction of the
 27:  * relationship is <em>from the subject to the object</em>.
 28:  *
 29:  * The vertices of this relatonship, the subject and object, are {@link NodeObject} instance
 30:  * references, while the relationship predicate is represented by a {@link TermObject}
 31:  * instance reference.
 32:  *
 33:  * The class features the following default offsets:
 34:  *
 35:  * <ul>
 36:  *  <li><tt>{@link kTAG_NID}</tt>: <em>Native identifier</em>. This required attribute holds
 37:  *      a <em>string</em> which represents the <em>combination of the subject, predicate and
 38:  *      object</em> of the relationship. This attribute must be managed with its offset,
 39:  *      although in derived classes it will be set automatically.
 40:  *  <li><tt>{@link kTAG_SUBJECT}</tt>: <em>Subject</em>. This attribute represents the
 41:  *      <em>origin of the relationship</em>, it is an <em>integer</em> value representing
 42:  *      the <em>reference to a {@link NodeObject} instance</em>. This attribute must be
 43:  *      managed with its offset.
 44:  *  <li><tt>{@link kTAG_PREDICATE}</tt>: <em>Predicate</em>. This attribute represents the
 45:  *      <em>type of relationship</em>, it is a <em>string</em> value representing the
 46:  *      <em>reference to a {@link TermObject} instance</em>. This attribute must be managed
 47:  *      with its offset.
 48:  *  <li><tt>{@link kTAG_OBJECT}</tt>: <em>Object</em>. This attribute represents the
 49:  *      <em>destination of the relationship</em>, it is an <em>integer</em> value
 50:  *      representing the <em>reference to a {@link NodeObject} instance</em>. This attribute
 51:  *      must be managed with its offset.
 52:  * </ul>
 53:  *
 54:  * The {@link __toString()} method will return the value stored in the native identifier,
 55:  * if set, or the computed native identifier, which is the concatenation of the subject,
 56:  * predicate and object references separated by the {@link kTOKEN_INDEX_SEPARATOR} token.
 57:  *
 58:  * Objects of this class feature a primary key which is not persistent: the vertices
 59:  * referenced in the native identifier are integer sequences which depend on the order these
 60:  * objects were inserted: this means that both {@link NodeObject} and {@link EdgeObject}
 61:  * instances must be re-created when exported.
 62:  *
 63:  * Objects of this class can hold any additional attribute that is considered necessary or
 64:  * useful to define and share the current node. In this class we define only those
 65:  * attributes that constitute the core functionality of the object, derived classes will add
 66:  * attributes specific to the domain in which the object will operate.
 67:  *
 68:  *  @author     Milko A. Škofič <m.skofic@cgiar.org>
 69:  *  @version    1.00 11/02/2014
 70:  */
 71: class EdgeObject extends OntologyObject
 72: {

 87:     /**
 88:      * <h4>Return global identifier</h4>
 89:      *
 90:      * The global identifier of the current object is represented by the subject, predicate
 91:      * and object references separated by the {@link kTOKEN_INDEX_SEPARATOR} token.
 92:      *
 93:      * @access public
 94:      * @return string               The global identifier.
 95:      */
 96:     public function __toString()
 97:     {
 98:         //
 99:         // Get relationship terms.
100:         //
101:         $terms = Array();
102:         $terms[] = $this->offsetGet( kTAG_SUBJECT );
103:         $terms[] = $this->offsetGet( kTAG_PREDICATE );
104:         $terms[] = $this->offsetGet( kTAG_OBJECT );
105:         
106:         return implode( kTOKEN_INDEX_SEPARATOR, $terms );                           // ==>
107:     
108:     } // __toString.

148:     /**
149:      * Handle offset and value before setting it
150:      *
151:      * In this class we cast the value of the relationship vertices into node reference, and
152:      * the value of the predicate into a term reference, if provided as objects; we also
153:      * ensure the provided objects arer of the correct type.
154:      *
155:      * @param reference             $theOffset          Offset reference.
156:      * @param reference             $theValue           Offset value reference.
157:      *
158:      * @access protected
159:      * @return mixed                <tt>NULL</tt> set offset value, other, return.
160:      *
161:      * @throws Exception
162:      *
163:      * @see kTAG_SUBJECT kTAG_PREDICATE kTAG_OBJECT
164:      */
165:     protected function preOffsetSet( &$theOffset, &$theValue )
166:     {
167:         //
168:         // Call parent method to resolve offset.
169:         //
170:         $ok = parent::preOffsetSet( $theOffset, $theValue );
171:         if( $ok === NULL )
172:         {
173:             //
174:             // Intercept node.
175:             //
176:             if( ($theOffset == kTAG_SUBJECT)
177:              || ($theOffset == kTAG_OBJECT) )
178:             {
179:                 //
180:                 // Handle objects.
181:                 //
182:                 if( is_object( $theValue ) )
183:                 {
184:                     //
185:                     // If term, get its reference.
186:                     //
187:                     if( $theValue instanceof NodeObject )
188:                         $theValue = $theValue->Reference();
189:                 
190:                     //
191:                     // If not a term, complain.
192:                     //
193:                     else
194:                         throw new \Exception(
195:                             "Unable to set edge vertex: "
196:                            ."provided an object other than a node." );          // !@! ==>
197:             
198:                 } // Object.
199:             
200:                 //
201:                 // Cast to integer.
202:                 //
203:                 else
204:                     $theValue = (int) $theValue;
205:             
206:             } // Setting tag.
207:             
208:             //
209:             // Intercept term.
210:             //
211:             if( $theOffset == kTAG_PREDICATE )
212:             {
213:                 //
214:                 // Handle objects.
215:                 //
216:                 if( is_object( $theValue ) )
217:                 {
218:                     //
219:                     // If term, get its reference.
220:                     //
221:                     if( $theValue instanceof TermObject )
222:                         $theValue = $theValue->Reference();
223:                 
224:                     //
225:                     // If not a term, complain.
226:                     //
227:                     else
228:                         throw new \Exception(
229:                             "Unable to set predicate: "
230:                            ."provided an object other than a term." );          // !@! ==>
231:             
232:                 } // Object.
233:             
234:                 //
235:                 // Cast to string.
236:                 //
237:                 else
238:                     $theValue = (string) $theValue;
239:             
240:             } // Setting term.
241:             
242:         } // Passed preflight.
243:         
244:         return $ok;                                                                 // ==>
245:     
246:     } // preOffsetSet.
